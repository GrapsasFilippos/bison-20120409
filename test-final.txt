int main()
{

  int i = 1;
  double a, b, p, fa, fb, fp, E;

  E = 0.001;
  n = 20;
  a = 0;
  b = 1;
  fp = 1;

  while(((b-a)/2>E) && (i<=n) && (fp != 0))
  {
  p=(a+b)/2;
  fa = a * a - 2;
  fb = b * b - 2;
  fp = p * p - 2;
  if(fa*fp<0)
   {
    b=p;
   }
   else
   {
    a=p;
   }
   i = i + 1;
  };
}
